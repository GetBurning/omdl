78 //! Map an index position into a circularly indexed list.
79 /***************************************************************************//**
80  \param i <integer> Any index, in or out of bounds.
81  \param l <integer> The circular list length.
82  \param f <integer> The starting index number.
83 
84  \returns <integer> A index position in the circular list within the
85  range <tt>[f : l+f-1]</tt>.
86 *******************************************************************************/
87 function circular_index
88 (
89  i,
90  l,
91  f = 0
92 ) = f + (((i % l) + l) % l);
