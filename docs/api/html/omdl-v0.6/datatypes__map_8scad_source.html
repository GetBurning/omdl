60 //! Return the index of a map key.
61 /***************************************************************************//**

63  \param k <string> A map key.
64 
65  \returns <integer> The index of the map entry if it exists.

68 function get_map_i
69 (

137 //! Perform some basic validation/checks on a map.
138 /***************************************************************************//**
139  \param m <matrix-2xN> A list of N key-value map pairs.
140 
141  \param verbose <boolean> Be verbose during check.
142 
143  \details
144 
145  Check that: (1) each entry has key-value 2-tuple, (2) each key is a
146  string, and (3) key identifiers are unique.
147 *******************************************************************************/
148 module map_check
149 (
150  m,
151  verbose = false
152 )
153 {
154  if (verbose) log_info("begin map check");
155 
156  if (verbose) log_info ("checking map format and keys.");
157 
158  if ( get_map_size(m) > 0 )
159  for ( i = [0:get_map_size(m)-1] )
160  {
161  entry = m[i];
162  key = first(entry);
163 
164  // each entry has key-value 2-tuple.
165  if ( 2 != len(entry) )
166  {
167  log_error
168  (
169  str (
170  "map index ", i,
171  ", entry=", entry,
172  ", has incorrect count=[", len(entry),"]"
173  )
174  );
175  }
176 
177  // each key must be a string.
178  if ( is_string(key) == false )
179  {
180  log_error
181  (
182  str (
183  "map index ", i,
184  ", entry=", entry,
185  ", key=[", key,"] is not a string."
186  )
187  );
188  }
189 
190  // no repeat key identifiers
191  if ( len(first(search([key], m, 0, 0))) > 1 )
192  log_warn
193  (
194  str(
195  "map index ", i,
196  ", key=[", key,"] not unique."
197  )
198  );
199  }
200 
201  if (verbose)
202  {
203  log_info
204  (
205  str (
206  "map size: ",
207  get_map_size(m), " entries."
208  )
209  );
210 
211  log_info("end map check");
212  }
213 }

215 //! Dump each map entry to the console.
216 /***************************************************************************//**
217  \param m <matrix-2xN> A list of N key-value map pairs.
218  \param sort <boolean> Sort the output by key.
219  \param number <boolean> Output index number.
220  \param p <integer> Number of places for zero-padded numbering.
221 *******************************************************************************/
222 module map_dump
223 (
224  m,
225  sort = true,
226  number = true,
227  p = 3
228 )
229 {
230  if ( get_map_size(m) > 0 )
231  {
232  keys = get_map_kl(m);
233  maxl = max( [for (i = keys) len(i)] ) + 1;
234 
235  for (key = sort ? qsort(keys) : keys)
236  {
237  idx = get_map_i(m, key);
238 
239  log_echo
240  (
241  str (
242  number ? chr(consts(p-len(str(idx)), 48)) : empty_str,
243  number ? str(idx, ": ") : empty_str,
244  chr(consts(maxl-len(key), 32)), "'", key, "' = ",
245  "'", get_map_v(m, key), "'"
246  )
247  );
248  }
249  }
250 
251  if ( number )
252  log_echo(str("map size: ", get_map_size(m), " entries."));
253 }
