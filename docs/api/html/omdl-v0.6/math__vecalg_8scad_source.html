68 //! Compute the distance between two Euclidean points.
69 /***************************************************************************//**
70  \param p1 <point> A point coordinate 1.
71  \param p2 <point> A point coordinate 2.
72 
73  \returns <decimal> The distance between the two points.
74  Returns \b undef when points do not have equal dimensions.
75 
76  \details
77 
78  When \p p2 is not given, it is assumed to be at the origin. This
79  function is similar to [norm].
80 
81  [norm]: https://en.wikibooks.org/wiki/OpenSCAD_User_Manual/Mathematical_Functions#norm
82 *******************************************************************************/
83 function distance_pp
84 (
85  p1,
86  p2
87 ) = let(d = len(p1))
88  !(d > 0) ? abs(p1 - defined_or(p2, 0))
89  : is_defined(p2) ?
90  sqrt(sum([for (i=[0:d-1]) (p1[i]-p2[i])*(p1[i]-p2[i])]))
91  : sqrt(sum([for (i=[0:d-1]) p1[i]*p1[i]]));
92 
93 //! Test if a point is left, on, or right of an infinite line in a Euclidean 2d-space.
94 /***************************************************************************//**
95  \param p1 <point-2d> A 2d point coordinate 1.
96  \param p2 <point-2d> A 2d point coordinate 2.
97  \param p3 <point-2d> A 2d point coordinate 3.
98 
99  \returns <decimal> (\b > 0) for \p p3 \em left of the line through
100  \p p1 and \p p2, (\b = 0) for p3 \em on the line, and
101  (\b < 0) for p3 right of the line.
102 
103  \details
104 
105  Function patterned after [Dan Sunday, 2012].
106 
107  [Dan Sunday, 2012]: http://geomalgorithms.com/a01-_area.html
108 *******************************************************************************/
109 function is_left_ppp
110 (
111  p1,
112  p2,
113  p3
114 ) = ((p2[0]-p1[0]) * (p3[1]-p1[1]) - (p3[0]-p1[0]) * (p2[1]-p1[1]));

203 //! Compute the dot product of two lines (or vectors).
204 /***************************************************************************//**
205  \param l1 <line> A n-dimensional line (or vector) 1.
206  \param l2 <line> A n-dimensional line (or vector) 2.
207 
208  \returns <decimal> The dot product of \p l1 with \p l2.
209  Returns \b undef when lines (or vectors) have different
210  dimensions.
211 
212  \details
213 
214  This function supports the abstraction outlined in \ref dt_line.
215  The built-in operation will be more efficient in situations that do
216  not make use of the aforementioned abstraction.
217 
218  See \ref dt_line for argument specification and conventions.
219  See [Wikipedia] for more information.
220 
221  [Wikipedia]: https://en.wikipedia.org/wiki/Dot_product
222 *******************************************************************************/
223 function dot_ll
224 (
225  l1,

229 //! Compute the cross product of two lines (or vectors) in a Euclidean 3d or 2d-space.
230 /***************************************************************************//**
231  \param l1 <line-3d|line-2d> A 3d or 2d line (or vector) 1.
232  \param l2 <line-3d|line-2d> A 3d or 2d line (or vector) 2.
233 
234  \returns <decimal|vector-2d> The cross product of \p l1 with \p l2.
235  Returns \b undef when lines (or vectors) have different
236  dimensions.
237 
238  \details
239 
240  This function supports the abstraction outlined in \ref dt_line.
241  The built-in operation will be more efficient in situations that do
242  not make use of the aforementioned abstraction.
243 
244  See \ref dt_line for argument specification and conventions.
245  See Wikipedia [cross] and [determinant] for more information.
246 
247  \note This function returns the 2x2 determinant for 2d vectors.
248 
249  [cross]: https://en.wikipedia.org/wiki/Cross_product
250  [determinant]: https://en.wikipedia.org/wiki/Determinant
251 *******************************************************************************/
252 function cross_ll
253 (
254  l1,
255  l2
256 ) = cross(get_line2origin(l1), get_line2origin(l2));

258 //! Compute the scalar triple product of three lines (or vectors) in a Euclidean 3d or 2d-space.
259 /***************************************************************************//**
260  \param l1 <line-3d|line-2d> A 3d or 2d line (or vector) 1.
261  \param l2 <line-3d|line-2d> A 3d or 2d line (or vector) 2.
262  \param l3 <line-3d|line-2d> A 3d or 2d line (or vector) 3.
263 
264  \returns <decimal|vector-2d> The scalar triple product.
265  Returns \b undef when lines (or vectors) have different
266  dimensions.
267 
268  \details
269 
270  [l1, l2, l3] = l1 * (l2 x l3)
271 
272  See \ref dt_line for argument specification and conventions.
273  See [Wikipedia] for more information.
274 
275  \warning Returns a 2d vector result for 2d vectors. The cross product
276  computes the 2x2 determinant of the vectors <tt>(l2 x l3)</tt>,
277  a scalar value, which is then \e multiplied by \c l1.
278 
279  [Wikipedia]: https://en.wikipedia.org/wiki/Triple_product
280 *******************************************************************************/
281 function striple_lll
282 (
283  l1,
284  l2,
285  l3
286 ) = (get_line2origin(l1) * cross_ll(l2, l3));

288 //! Compute the angle between two lines (or vectors) in a Euclidean 3d or 2d-space.
289 /***************************************************************************//**
290  \param l1 <line-3d|line-2d> A 3d or 2d line (or vector) 1.
291  \param l2 <line-3d|line-2d> A 3d or 2d line (or vector) 2.
292 
293  \returns <decimal> The angle between the two lines (or vectors) in
294  degrees. Returns \b undef when lines (or vectors) have
295  different dimensions or when they do not intersect.
296 
297  \details
298 
299  See \ref dt_line for argument specification and conventions.
300 
301  \note For 3d lines (or vectors), a normal is required to uniquely
302  identify the perpendicular plane and axis of rotation. This
303  function calculates the positive angle, and the plane and
304  axis of rotation will be that which fits this assumed
305  positive angle.
306 
307  \sa angle_lll().
308 *******************************************************************************/
309 function angle_ll
310 (
311  l1,
312  l2
313 ) = let(d = get_line_dim(l1))
314  (d == 2) ? atan2(cross_ll(l1, l2), dot_ll(l1, l2))
315  : (d == 3) ? atan2(distance_pp(cross_ll(l1, l2)), dot_ll(l1, l2))
316  : undef;

324  \returns <decimal> The angle between the two lines (or vectors) in

327 
328  \details

330  See \ref dt_line for argument specification and conventions.

389 //! Convert a planes' normal specification into a normal vector.
390 /***************************************************************************//**
391  \param pn <pnorm> A plane normal \ref dt_pnorm "specification".
392 
393  \param cw <boolean> Point ordering. When the plane specified as
394  non-collinear points, this indicates ordering.
395 
396  \returns <normal> A vector-3d normal to the plane.
397 
398  \details
399 
400  See \ref dt_pnorm for argument specification and conventions.
401 *******************************************************************************/
402 function get_pnorm2nv
403 (
404  pn,
405  cw = true
406 ) = not_defined(len(pn[0])) ?
407  (
408  (len(pn) == 3) ? pn : (len(pn) == 2) ? [pn[0], pn[1], 0]: undef
409  )
410  : let
411  (
412  q = [for (i=pn) (len(i) == 3) ? i : (len(i) == 2) ? [i[0], i[1], 0]: undef]
413  )
414  (len(pn) == 1) ? q[0]
415  : (len(pn) == 2) ? cross(q[0], q[1])
416  : cross(q[0]-q[1], q[2]-q[1]) * ((cw == true) ? 1 : -1);
