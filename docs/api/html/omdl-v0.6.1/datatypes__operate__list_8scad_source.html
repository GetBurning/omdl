585 //! Sort the numeric or string elements of a list using quick sort.
586 /***************************************************************************//**
587  \param v <number-list|string-list> A list of values.
588  \param i <integer> The sort column index for iterable elements.
589  \param r <boolean> Reverse the sort order.
590 
591  \returns <list> A list with elements sorted in ascending order.
592  Returns \b undef when \p v is not defined or is not a list.
593 
594  \details
595 
596  \warning This implementation relies on the comparison operators
597  '<' and '>' which expect the operands to be either two scalar
598  numbers or two strings. Therefore, this function will not
599  correctly sort lists elements that are not numbers or
600  strings. Elements with unknown order are placed at the end
601  of the list.
602 
603  See [Wikipedia] for more information.
604 
605  [Wikipedia]: https://en.wikipedia.org/wiki/Quicksort
606 *******************************************************************************/
607 function qsort
608 (
609  v,
610  i,
611  r = false
612 ) = not_defined(v) ? undef
613  : !is_list(v) ? undef
614  : is_empty(v) ? empty_lst
615  : let
616  (
617  mp = v[floor(len(v)/2)],
618  me = not_defined(i) ? mp : mp[i],
619 
620  lt = [for (j = v) let(k = not_defined(i) ? j : j[i]) if (k < me) j],
621  eq = [for (j = v) let(k = not_defined(i) ? j : j[i]) if (k == me) j],
622  gt = [for (j = v) let(k = not_defined(i) ? j : j[i]) if (k > me) j],
623 
624  ou =
625  [
626  for (j = v) let(k = not_defined(i) ? j : j[i])
627  if ( !((k < me) || (k == me) || (k > me)) ) j
628  ],
629 
630  sp = (r == true) ?
631  concat(qsort(gt, i, r), eq, qsort(lt, i, r), ou)
632  : concat(qsort(lt, i, r), eq, qsort(gt, i, r), ou)
633  )
634  sp;

636 //! Hierarchically sort an arbitrary data list using quick sort.
637 /***************************************************************************//**
638  \param v <data> A list of values.
639  \param i <integer> The sort column index for iterable elements.
640  \param d <integer> The recursive sort depth.
641  \param r <boolean> Reverse the sort order.
642  \param s <boolean> Order ranges by their numerical sum.
643 
644  \returns <list> With all elements sorted in ascending order.
645  Returns \b undef when \p v is not defined or is not a list.
646 
647  \details
648 
649  Elements are ordered using compare(). See its documentation for a
650  description of the parameter \p s. To recursively sort all
651  elements, set \p d greater than, or equal to, the maximum level of
652  hierarchy in \p v.
653 
654  See [Wikipedia] for more information.
655 
656  [Wikipedia]: https://en.wikipedia.org/wiki/Quicksort
657 *******************************************************************************/
658 function qsort2
659 (
660  v,
661  i,
662  d = 0,
663  r = false,
664  s = true
665 ) = not_defined(v) ? undef
666  : !is_list(v) ? undef
667  : is_empty(v) ? empty_lst
668  : let
669  (
670  mp = v[floor(len(v)/2)],
671  me = not_defined(i) ? mp : mp[i],
672 
673  lt =
674  [
675  for (j = v)
676  let(k = not_defined(i) ? j : j[i])
677  if (compare(me, k, s) == -1)
678  ((d > 0) && is_list(k)) ? qsort2(k, i, d-1, r, s) : j
679  ],
680  eq =
681  [
682  for (j = v)
683  let(k = not_defined(i) ? j : j[i])
684  if (compare(me, k, s) == 0)
685  ((d > 0) && is_list(k)) ? qsort2(k, i, d-1, r, s) : j
686  ],
687  gt =
688  [
689  for (j = v)
690  let(k = not_defined(i) ? j : j[i])
691  if (compare(me, k, s) == +1)
692  ((d > 0) && is_list(k)) ? qsort2(k, i, d-1, r, s) : j
693  ],
694  sp = (r == true) ?
695  concat(qsort2(gt, i, d, r, s), eq, qsort2(lt, i, d, r, s))
696  : concat(qsort2(lt, i, d, r, s), eq, qsort2(gt, i, d, r, s))
697  )
698  (d > 0) ? qsort2(sp, i, d-1, r, s) : sp;
